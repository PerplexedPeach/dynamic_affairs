			LISTNAME = affairs_pool

		save_scope_value_as = {
			name = min_gold_cost
			value = start_affair_cost
		}

			limit = {
				has_trait_xp = {
					trait = lifestyle_prostitute

			save_scope_value_as = {
				name = gold_cost
				value = {
					add = scope:min_gold_cost
					multiply = 2.0
				}
			}
		}
		else_if = {
			limit = {
				has_trait_xp = {
					trait = lifestyle_prostitute
					value >= 50
				}
			}
			save_scope_value_as = {
				name = gold_cost
				value = {
					add = scope:min_gold_cost
					multiply = 1.25
				}
			}
		}
		else_if = {
			limit = {
				has_trait = lifestyle_prostitute
			}
			save_scope_value_as = {
				name = gold_cost
				value = {
					add = scope:min_gold_cost
					multiply = 1.5
				}
			}
		}
		else = {
			save_scope_value_as = {
				name = gold_cost
				value = scope:min_gold_cost
			}

			weight = {
				base = 1
				# add based on attraction, up to +100%
				modifier = {
					add = {
						value = attraction
						divide = high_positive_attraction
					}
				}
				# more likely if they have extra coins to spare
				modifier = {
					factor = 2.5
					gold >= scope:gold_cost_extra
				}

				# however, it's still more likely than not since they know you're open for it, so it's not as big of a factor
				modifier = {
					add = 0.4

						target = root
						modifier = dominant_opinion
					}
				}
			}
			save_scope_as = affairs_partner
		}

	option = {

			has_trait = lifestyle_prostitute

				trait = lifestyle_prostitute
				value < 50
			}
		}
		scope:affairs_partner = {
			pay_short_term_gold = {
				target = root
				gold = scope:gold_cost
			}
		}
		reverse_add_opinion = {
			target = scope:affairs_partner
			modifier = serviced_opinion
			opinion = 5
		}
		trait = lifestyle_prostitute
		trigger_event = LIDA.101
	}
	option = {

				trait = lifestyle_prostitute
				value >= 50
			}

				trait = lifestyle_prostitute
				value < 100
			}
		}
		scope:affairs_partner = {
			pay_short_term_gold = {
				target = root
				gold = scope:gold_cost
			}
		}
		reverse_add_opinion = {
			target = scope:affairs_partner
